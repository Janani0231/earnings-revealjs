from math import pow

def compound_growth(pv: float, r: float, n: int, t: float) -> float:
    """Future Value with compounding
    pv: principal, r: annual rate, n: compounds/yr, t: years
    """
    return pv * pow(1 + r / n, n * t)

print(compound_growth(1_000_000, 0.08, 4, 1))
